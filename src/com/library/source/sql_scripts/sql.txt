1. You need to create a database for this system.
CREATE DATABASE library_system;

2. Let's create a table named persons to store both members and admins
CREATE TABLE persons (
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    cid VARCHAR(12) NOT NULL,
    dob DATE NOT NULL,
    phone VARCHAR(15) NOT NULL,
    address_id int NOT NULL,
    PRIMARY KEY (cid),
    FOREIGN KEY (address_id) REFERENCES addresses(address_id));

3. Create a table named addresses to store everyone's addresses.
    CREATE TABLE addresses (
        address_id int GENERATED ALWAYS AS IDENTITY,
        street VARCHAR(95) NOT NULL,
        city VARCHAR(35) NOT NULL,
        state_id int NOT NULL,
        zipcode VARCHAR(5) NOT NULL,
        PRIMARY KEY(address_id),
        FOREIGN KEY(state_id) REFERENCES states(state_id));

4. Create a table name countries to store countries' name.
    CREATE TABLE countries (
        country_id SMALLINT GENERATED ALWAYS AS IDENTITY,
        country_name VARCHAR(60) NOT NULL,
        iso_code VARCHAR(2) NOT NULL,
        PRIMARY KEY (country_id));

5. Create a table named states to store state name.
    CREATE TABLE states (
        state_id int
        country_id int
        state_name VARCHAR(60) NOT NULL,
        PRIMARY KEY (state_id),
        FOREIGN KEY (country_id) REFERENCES countries(country_id));

4. Create a table named accounts to store librarians and members' accounts.
    CREATE TABLE accounts (
        id VARCHAR(7),
        cid VARCHAR(12) NOT NULL,
        password TEXT NOT NULL,
        status SMALLINT NOT NULL,
        PRIMARY KEY (id),
        UNIQUE (cid),
        FOREIGN KEY (cid) REFERENCES persons(cid));
